#!/bin/bash

# This is a simple bash script

# Variables
NAME="World"
CURRENT_DATE=$(date)

# Function to greet
greet() {
    echo "Hello, $1!"
}

# Main script
echo "Starting the script..."
greet "$NAME"

# Display current date
echo "Today is: $CURRENT_DATE"

# Simple arithmetic
num1=10
num2=5
result=$((num1 + num2))
echo "The sum of $num1 and $num2 is: $result"

# Exit with success status
exit 0
